export const CourseLevel = ({children, level}) => {
    const filteredItems = items.filter(item => item.level == level)

                    actions={item.actions}
                    test_case={item.test_case}>{item.title}</CourseItem>
            </li>
        ));
    return <ol>{filteredItems}</ol>;
};

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

export const items = [
    {
        "level": 1,
        "title": "Dummy",
        "youtube": "",
        "test_case": "",
        "actions": ""
    },
];

export const CourseItem = ({children, youtube, actions, test_case}) => (
    <div style={{
        paddingTop: 8,
        paddingBottom: 8,
    }}>
        <strong><a href={youtube ? youtube : "#"} target={youtube ? "_blank" : ""}>{children}</a></strong>
        <br/>
        <strong>Actions</strong>: {actions ? actions : "N/A"}
        <br/>
        <strong>Sample Test Case</strong>: {test_case ? test_case : "N/A"}
    </div>
);

export const CourseLevel = ({children, level}) => {
    const filteredItems = items.filter(item => item.level == level)
        .map(item => (
            <li key={item.youtube}>
                <CourseItem
                    youtube={item.youtube}
                    actions={item.actions}
                    test_case={item.test_case}>{item.title}</CourseItem>
            </li>
        ));
    return <ol>{filteredItems}</ol>;
};

<Tabs
    defaultValue="level_1"
    values={[
        {label: 'Level 1', value: 'level_1'},
        {label: 'Level 2', value: 'level_2'},
        {label: 'Level 3', value: 'level_3'},
        {label: 'Level 4', value: 'level_4'},
    ]}>
    <TabItem value="level_1"><CourseLevel level={1}/></TabItem>
    <TabItem value="level_2"><CourseLevel level={2}/></TabItem>
    <TabItem value="level_3"><CourseLevel level={3}/></TabItem>
    <TabItem value="level_4"><CourseLevel level={4}/></TabItem>
</Tabs>
